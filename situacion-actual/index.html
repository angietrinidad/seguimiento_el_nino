## Cota actual del río Paraguay

```{python}
#| echo: false
import json
from IPython.display import HTML
d = json.load(open("datos/nivel-rio-actual.json", encoding="utf-8"))
crit, pico, esc = d["critica_asuncion_m"], 7.84, 8.0
asu = next((e for e in d["estaciones"] if e["nombre"] == "Asunción"), None)

def flecha(c):
    if c is None: return ""
    if c > 0: return f"<span style='color:#8a1c1c'>▲ +{c} cm</span>"
    if c < 0: return f"<span style='color:#1b5e7a'>▼ {c} cm</span>"
    return "= 0 cm"

def pos(v): return min(v / esc, 1) * 100

barra = ""
if asu:
    nivel = asu["nivel_m"]; margen = round(crit - nivel, 2)
    barra = f"""
<div style="max-width:560px">
  <div style="font-size:15px;margin-bottom:.2em">Asunción: <b style="font-size:22px">{nivel} m</b>
    &nbsp;{flecha(asu['cambio_cm'])}
    &nbsp;<span style="color:#555">— a {margen} m del nivel crítico</span></div>
  <div style="position:relative;height:30px;background:#e8eef2;border-radius:6px;margin:1.6em 0 .4em">
    <div style="position:absolute;left:0;top:0;height:100%;width:{pos(nivel):.1f}%;background:#2b6cb0;border-radius:6px 0 0 6px"></div>
    <div style="position:absolute;left:{pos(crit):.1f}%;top:-6px;height:42px;border-left:2px dashed #b26a00"></div>
    <div style="position:absolute;left:{pos(crit):.1f}%;top:-22px;font-size:11px;color:#b26a00;transform:translateX(-50%)">crítico 5,5 m</div>
    <div style="position:absolute;left:{pos(pico):.1f}%;top:-6px;height:42px;border-left:2px dashed #8a1c1c"></div>
    <div style="position:absolute;left:{pos(pico):.1f}%;top:-22px;font-size:11px;color:#8a1c1c;transform:translateX(-50%)">pico 2016: 7,84 m</div>
  </div>
  <div style="font-size:11px;color:#888">0 m &nbsp;&nbsp;&nbsp; escala hasta 8 m</div>
</div>"""

filas = "".join(
    f"<tr><td style='padding:3px 10px 3px 0'>{e['nombre']}</td>"
    f"<td style='padding:3px 10px;text-align:right'><b>{e['nivel_m']} m</b></td>"
    f"<td style='padding:3px 0;text-align:right'>{flecha(e['cambio_cm'])}</td></tr>"
    for e in d["estaciones"])
tabla = (f"<table style='font-size:14px;border-collapse:collapse;margin-top:1em'>"
         f"<tr style='color:#555;border-bottom:1px solid #ddd'><th style='text-align:left'>Estación (norte→sur)</th>"
         f"<th style='text-align:right'>Nivel</th><th style='text-align:right'>Variación</th></tr>{filas}</table>")
pie = (f"<div style='font-size:12px;color:#888;margin-top:.8em'>Cota del <b>{d['fecha']}</b> · "
       f"fuente: DINAC (meteorologia.gov.py), actualizada a diario.</div>")
HTML(barra + tabla + pie)
```

## Lluvia actual (estaciones automáticas de la DMH)

La amenaza **pluvial** (raudales por lluvia) no depende del río sino de la
intensidad de la lluvia. Este panel lee las **Estaciones Meteorológicas Automáticas
(EMAs)** de la DMH en el área metropolitana [@dmh_emas], con el acumulado del día,
y lo ubica en la [escala de colapso pluvial](../04-asuncion-metropolitana/index.qmd#una-escala-empírica-cuánta-lluvia-colapsa-asunción).

```{python}
#| echo: false
import json, os
from IPython.display import HTML
p = "datos/estaciones-dmh-metro.json"
if not os.path.exists(p):
    salida = "<div class='vacio'>Datos de estaciones DMH pendientes de generar.</div>"
else:
    d = json.load(open(p, encoding="utf-8"))
    activas = [e for e in d["estaciones"] if e["activa"] and e["precip_dia_mm"] is not None]
    inactivas = [e for e in d["estaciones"] if not e["activa"]]
    # nivel de la escala según el acumulado del día (bandas coherentes con 04)
    def banda(mm):
        if mm >= 150: return ("4 · Colapso / catástrofe", "#7a1f2b")
        if mm >= 80:  return ("3 · Anegamiento urbano extendido", "#d1495b")
        if mm >= 40:  return ("1-2 · Encharcamientos y raudales localizados", "#e8842c")
        if mm > 0:    return ("bajo el umbral de afectación", "#2a9d3f")
        return ("sin lluvia significativa (0 mm)", "#2a9d3f")
    maxmm = max((e["precip_dia_mm"] for e in activas), default=None)
    if maxmm is None:
        estado = "<i>sin estaciones activas con dato de lluvia ahora mismo.</i>"
    else:
        etq, col = banda(maxmm)
        estado = (f"Máximo acumulado hoy entre las estaciones activas: "
                  f"<b style='font-size:20px'>{maxmm} mm</b> → "
                  f"<b style='color:{col}'>{etq}</b>")
    filas = "".join(
        f"<tr><td style='padding:3px 12px 3px 0'>{e['ciudad']} — {e['nombre']}</td>"
        f"<td style='padding:3px 0;text-align:right'><b>{e['precip_dia_mm']} mm</b></td></tr>"
        for e in activas)
    tabla = (f"<table style='font-size:14px;border-collapse:collapse;margin:.6em 0'>"
             f"<tr style='color:#555;border-bottom:1px solid #ddd'>"
             f"<th style='text-align:left'>Estación activa</th>"
             f"<th style='text-align:right'>Acum. del día</th></tr>{filas}</table>")
    upd = (d.get("actualizado") or {}).get("local", "s/f")
    pie = (f"<div style='font-size:12px;color:#888;margin-top:.4em'>Última actualización del feed: "
           f"<b>{upd}</b> · fuente: DMH/DINAC (EMAs), orientativa, no oficial · "
           f"{len(inactivas)} estación(es) del metro sin transmitir al día. "
           f"Para refrescar: correr <code>scripts/estaciones_dmh.py</code>.</div>")
    salida = f"<div style='max-width:600px'><div style='margin-bottom:.4em'>{estado}</div>{tabla}{pie}</div>"
# HTML() debe ir a nivel superior (dentro de if/else no se auto-muestra)
HTML(salida)
```

::: {.callout-note appearance="minimal"}
La DMH publica la precipitación como **acumulado del día en hora local** (medición
cada 10 min); es orientativa. El acumulado diario **subestima el riesgo de raudal**,
que depende de la intensidad en minutos (ver la
[IDF simplificada](../04-asuncion-metropolitana/index.qmd#el-eje-de-duración-una-idf-simplificada)):
un valor moderado caído en una hora puede ya generar raudales.
:::
